	function continuar( url, cns )
	{
		frm.action = url;
		frm.cns.value = cns;
		frm.etapa.value = '';
		frm.url.value = '';
		frm.submit();
	}

	function nova_pesquisa()
	{
		frm.etapa.value = '';
		frm.url.value = '';
		frm.standalone.value = '1';
		frm.submit();
